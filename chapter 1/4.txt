/* 4. Write a program to implement the Scope and Shadowing .*/

fn main() {
    let x = 5;
    println!("Outer x = {}", x);

    {
        let x = 10;
        println!("Inner x (shadowed) = {}", x);

        {
            let x = 15;
            println!("Innermost x (shadowed again) = {}", x);
        }

        println!("Back to inner scope x = {}", x);
    }

    println!("Outer x after inner scopes = {}", x);
}
