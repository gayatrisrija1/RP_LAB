/* Write a program to implement the following
a. Implicit type declaration
b. Explicit type declaration
*/
fn main() {
    let a = 42;
    let b = 3.14;
    let c = "Hello";

    println!("Implicit types: a = {}, b = {}, c = {}", a, b, c);

    let x: i64 = 100;
    let y: f32 = 9.81;
    let z: &str = "Rust";

    println!("Explicit types: x = {}, y = {}, z = {}", x, y, z);
}
