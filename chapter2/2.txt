/* 2. Write a program to implement Borrowing and Dereferencing Operators*/

fn main() {
    let x = 10;
    
    let y = &x; 
    
    println!("Value of x: {}", x);
    println!("Value of y (borrowed reference): {}", y);
    
    let z = *y;  
    
    println!("Value of z (after dereferencing y): {}", z);
}
